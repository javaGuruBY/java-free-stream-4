public class HumanDemo {
    public static void main (String[] args) {


        Human h3 = new Human("Bill", 13);
        h3.greet();

        Human h4 = new Human("Vlad", 20);
        h4.greet();

    }
}
class Human {
    String name;
    int age;


     Human (String name, int age)
     {
         this.age = age;
         this.name = name;
     }


    void greet() {
        System.out.println("Hi, my name is " + name + ". I'm " + age);
    }
}
